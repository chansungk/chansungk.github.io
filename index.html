Given :

  • $L_t, H_t, C_t, V_t = t\,,\, ,\, ,\, $
  • $n = \,$

    : $\Delta C = (C_t - C_{t-1})\, /\, C_t-1$

    : $\Delta V = (V_t - V_{t-1})\, /\, V_t-1$

    MFM : $\frac{(C_t - L_t)-(H_t - C_t)}{H_t - L_t} \times V_t$

    MOMENTUM : $C_t - C_{t-(n-1)}$

    SMA : $\frac{C_t + C_{t-1}+...+C_{t-(n-1)}}{n}$

    WMA : $\frac{nC_t + (n-1)C_{t-1}+...+C_{t-(n-1)}}{n+(n-1)+...+1}$

    : $\frac{C_t}{SMA} \times 100$

def make_dataset(data, label, window_size):
    x_list = []
    y_list = []
    for i in range(len(data) - window_size):
        x_list.append(np.array((data.iloc[(i+1):(i+1)+window_size])).flatten().tolist()) # 2  
    
        #      (+)  [1, 0], 0  (-)  [0, 1] 
        if (label.iloc[i] - label.iloc[i+1]) > 0:
            y_list.append([1])
        else:
            y_list.append([0])
    return pd.DataFrame(x_list), pd.DataFrame(y_list)
